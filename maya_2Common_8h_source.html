102 #define AL_MAYA_CHECK_ERROR(status, ErrorString) { \
103  MStatus _status_##__LINE__ = status; \
104  if (!_status_##__LINE__) \
105  { \
106  MString maya_error_string = __FILE__ ":"; \
107  maya_error_string += __LINE__; \
108  maya_error_string += " "; \
109  maya_error_string += _status_##__LINE__.errorString(); \
110  maya_error_string += " : "; \
111  maya_error_string += ErrorString; \
112  MGlobal::displayError(maya_error_string); \
113  return status; \
114  } }
