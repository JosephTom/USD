219  static inline timespec timeDiff(const timespec startTime, const timespec endTime)
220  {
221  if (endTime.tv_nsec < startTime.tv_nsec)
222  {
223  return (timespec) {
224  tv_sec: endTime.tv_sec - 1 - startTime.tv_sec,
225  tv_nsec: 1000000000 + endTime.tv_nsec - startTime.tv_nsec
226  };
227  }
228  return (timespec) {
229  tv_sec: endTime.tv_sec - startTime.tv_sec,
230  tv_nsec: endTime.tv_nsec - startTime.tv_nsec
231  };
232  }
233  static inline timespec timeAdd(timespec t1, timespec t2)
234  {
235  int32_t sec = t2.tv_sec + t1.tv_sec;
236  int32_t nsec = t2.tv_nsec + t1.tv_nsec;
237  if (nsec >= 1000000000)
238  {
239  nsec -= 1000000000;
240  sec++;
241  }
242  return (timespec){ tv_sec: sec, tv_nsec: nsec };
243  }
