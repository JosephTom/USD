109  inline void clear()
110  { m_selected.clear(); }
111 
114  inline void add(SdfPath path)
115  {
116  if(std::find(m_selected.begin(), m_selected.end(), path) == m_selected.end())
117  {
118  m_selected.push_back(path);
119  }
120  }
121 
124  inline void remove(SdfPath path)
125  {
126  auto it = std::find(m_selected.begin(), m_selected.end(), path);
127  if(it != m_selected.end())
128  {
129  m_selected.erase(it);
130  }
131  }
132 
135  inline void toggle(SdfPath path)
136  {
137  auto it = std::find(m_selected.begin(), m_selected.end(), path);
138  if(it == m_selected.end())
139  {
140  m_selected.push_back(path);
141  }
142  else
143  {
144  m_selected.erase(it);
145  }
146  }
147 
150  inline bool isSelected(const SdfPath& path) const
151  { return std::find(m_selected.begin(), m_selected.end(), path) != m_selected.end(); }
152 
155  inline const SdfPathVector& paths() const
156  { return m_selected; }
157 
160  inline size_t size() const
161  { return m_selected.size(); }
162 
163 private:
164  SdfPathVector m_selected;
165 };
