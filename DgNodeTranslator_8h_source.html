333 template<typename T>
334 MStatus DgNodeTranslator::setVec3Anim(MObject node, MObject attr, const UsdGeomXformOp op, double conversionFactor)
335 {
336  MPlug plug(node, attr);
337  MStatus status;
338  const char* const xformErrorCreate = "DgNodeTranslator:setVec3Anim error creating animation curve";
339 
340  MFnAnimCurve acFnSetX;
341  acFnSetX.create(plug.child(0), NULL, &status);
342  AL_MAYA_CHECK_ERROR(status, xformErrorCreate);
343 
344  MFnAnimCurve acFnSetY;
345  acFnSetY.create(plug.child(1), NULL, &status);
346  AL_MAYA_CHECK_ERROR(status, xformErrorCreate);
347 
348  MFnAnimCurve acFnSetZ;
349  acFnSetZ.create(plug.child(2), NULL, &status);
350  AL_MAYA_CHECK_ERROR(status, xformErrorCreate);
351 
352  std::vector<double> times;
353  op.GetTimeSamples(&times);
354 
355  const char* const xformErrorKey = "DgNodeTranslator:setVec3Anim error setting key on animation curve";
356 
357  T value(0);
358  for(auto const& timeValue: times)
359  {
360  const bool retValue = op.GetAs<T>(&value, timeValue);
361  if (!retValue) continue;
362 
363  MTime tm(timeValue, MTime::kFilm);
364 
365  switch (acFnSetX.animCurveType())
366  {
367  case MFnAnimCurve::kAnimCurveTL: // time->distance: translation
368  case MFnAnimCurve::kAnimCurveTA: // time->angle: rotation
369  case MFnAnimCurve::kAnimCurveTU: // time->double: scale
370  {
371  acFnSetX.addKey(tm, value[0] * conversionFactor, MFnAnimCurve::kTangentGlobal, MFnAnimCurve::kTangentGlobal, NULL, &status);
372  AL_MAYA_CHECK_ERROR(status, xformErrorKey);
373  acFnSetY.addKey(tm, value[1] * conversionFactor, MFnAnimCurve::kTangentGlobal, MFnAnimCurve::kTangentGlobal, NULL, &status);
374  AL_MAYA_CHECK_ERROR(status, xformErrorKey);
375  acFnSetZ.addKey(tm, value[2] * conversionFactor, MFnAnimCurve::kTangentGlobal, MFnAnimCurve::kTangentGlobal, NULL, &status);
376  AL_MAYA_CHECK_ERROR(status, xformErrorKey);
377  break;
378  }
379  default:
380  {
381  break;
382  }
383  }
384  }
385 
386  return MS::kSuccess;
387 }
