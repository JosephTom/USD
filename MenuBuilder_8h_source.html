133  template<typename FnPlugin>
134  static MStatus generatePluginUI(FnPlugin& fnPlugin, const MString& prefix, const MString& extraOnInit = "", const MString& extraOnExit = "")
135  {
136  if(m_menus.empty())
137  return MS::kSuccess;
138  MString ui_init = prefix + "_initGUI";
139  MString ui_exit = prefix + "_exitGUI";
140 
141  // set up init script
142  std::ostringstream initGUI;
143  initGUI << "global proc " << ui_init.asChar() << "()\n{\n global string $gMainWindow;\n";
144 
145  if(extraOnInit.length())
146  initGUI << " " << extraOnInit.asChar() << std::endl;
147 
148  // set up exit script
149  std::ostringstream exitGUI;
150  exitGUI << "global proc " << ui_exit.asChar() << "()\n{\n";
151  if(extraOnExit.length())
152  exitGUI << " " << extraOnExit.asChar() << std::endl;
153  exitGUI << " deleteUI ";
154 
155  // now construct all of the menu related gubbins.
156  for(auto it = m_menus.begin(); it != m_menus.end(); ++it)
157  {
158  it->generate(initGUI, exitGUI, prefix.asChar());
159  }
160  m_menus.clear();
161 
162  // finish off
163 
164  initGUI << "}\n\n";
165  exitGUI << ";\n}\n\n";
166 
167  MGlobal::executeCommand(initGUI.str().c_str());
168  MGlobal::executeCommand(exitGUI.str().c_str());
169 
170  #if AL_USD_PRINT_UI_CODE
171  std::cout << initGUI.str() << std::endl;
172  std::cout << exitGUI.str() << std::endl;
173  #endif
174 
175  return fnPlugin.registerUI(ui_init, ui_exit);
176  }
177 
178 private:
179  static std::set<Menu> m_menus;
180 };
