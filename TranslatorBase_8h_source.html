271 #define AL_USDMAYA_DEFINE_TRANSLATOR(PlugClass, TranslatedType) \
272 TfRefPtr<PlugClass> \
273 PlugClass::create(TranslatorContextPtr context) { \
274  TfType const &type = TfType::Find<TranslatedType>(); \
275  if(not type.IsUnknown()) { \
276  TfRefPtr<PlugClass> plugin = TfCreateRefPtr(new This()); \
277  plugin->setTranslatedType(type); \
278  plugin->setContext(context); \
279  if(!plugin->initialize()) return TfRefPtr<PlugClass>(); \
280  return plugin; \
281  } \
282  else { \
283  TF_CODING_ERROR( \
284  "Failed to get %s usd type, maybe the needed plugin is not loaded", \
285  typeid(TranslatedType).name()); \
286  return TfNullPtr; \
287  } \
288 } \
289  \
290 TF_REGISTRY_FUNCTION(TfType) \
291 { \
292  TfType::Define<PlugClass, TfType::Bases<TranslatorBase>>() \
293  .SetFactory<TranslatorFactory<PlugClass>>(); \
