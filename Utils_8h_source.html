95 struct guid_compare
96 {
97  inline bool operator () (const i128 a, const i128 b) const
98  {
99  const uint32_t lt_mask = movemask16i8(cmplt16i8(a, b));
100  const uint32_t eq_mask = 0xFFFF & (~movemask16i8(cmpeq16i8(a, b)));
101  if(!eq_mask) return false;
102 
103  // find first bit value that is not equal
104  const uint32_t index = __builtin_ctz(eq_mask);
105  // now see whether that bit has been set
106  return (lt_mask & (1 << index)) != 0;
107  }
108 };
109 
110 #else
111 
114 struct guid_compare
115 {
120  inline bool operator () (const guid& a, const guid& b) const
121  {
122  for(int i = 0; i < 16; ++i)
123  {
124  if(a.uuid[i] < b.uuid[i]) return true;
125  if(a.uuid[i] > b.uuid[i]) return false;
126  }
127  return false;
128  }
129 };
130 #endif
